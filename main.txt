package placyk;
import javax.swing.*;
import java.awt.*;
import java.awt.geom.Ellipse2D;
import java.util.Random;

//Glowna klasa programu - ramka
public class PFrame extends JFrame
{
	public PRunnable r[];
	public Thread t[];
	public int size;
	
	//konstruktor glownej klasy programu - ramki
	public PFrame()
	{
		add(new ImageComponent());
		pack();
		
		size = 17;
		r = new PRunnable[20];
		t = new Thread[20];
		Random rnd = new Random();
		
		for(int i=0;i<20;i++)
		{
			r[i] = new PRunnable(this,rnd.nextInt(390)+40,rnd.nextInt(520)+50);
			t[i] = new Thread(r[i]);
			t[i].start();
			t[i].setPriority(1);
		}
	}
	
	//utworzenie ramki
	public static void main(String[] args)
	{
		EventQueue.invokeLater(new Runnable()
		{
			public void run()
			{
				PFrame frame = new PFrame();
				frame.setTitle("Placyk by SlimSzady & Michu4k");
				frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
				frame.setLocationByPlatform(true);
				frame.setResizable(false);
				frame.setVisible(true);
			}
		});
	}
	
	private void checkCollisions()
	{
		for(int i=0;i<20;i++)
			for(int j=i+1;j<20;j++)
			{
				if( Math.abs(r[i].getX() - r[j].getX() )<=size && Math.abs(r[i].getY() - r[j].getY() )<=size )
				{
					r[i].setCollision(j);
					r[j].setCollision(i);
				}
				else
				{
					r[i].clearCollision(j);
					r[j].clearCollision(i);
				}
			}
	}
	
	//Komponent na ktorym wszystko jest malowane
	class ImageComponent extends JComponent
	{
		//szerokosc ekranu
		private final int WIDTH = 470;
		//wysokosc ekranu
		private final int HEIGHT = 630;
		
		//rysowanie komponentu
		public void paintComponent(Graphics g)
		{	
			Graphics2D g2 = (Graphics2D) g;
			Image image = new ImageIcon("Placyk.png").getImage();
			
			//obraz
			g2.drawImage(image, 0, 0, null);
			
			//podpis
			g2.setColor(Color.BLACK);
			g2.setFont(new Font("Century", Font.BOLD, 12));
			g2.drawString("Michal Krysztofik", 20, HEIGHT+5);
			g2.drawString("Pawel Kowalczyk", WIDTH-120, HEIGHT+5);
			
			//ludziki
			for(int i=0;i<20;i++)
			{
				g2.setColor(Color.BLACK);
				g2.fill(new Ellipse2D.Double(r[i].getX(), r[i].getY(), size, size));
				
				g2.setColor(r[i].getC());
				g2.fill(new Ellipse2D.Double(r[i].getX(), r[i].getY(), size-2, size-2));
			}
			
			checkCollisions();
		}
		
		//zwraca wymiary komponentu
		public Dimension getPreferredSize(){return new Dimension(WIDTH, HEIGHT);}
	}
}
