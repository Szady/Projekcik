package placyk;

//Nowy watek
public class PRunnable implements Runnable
{
	private PFrame frame;
	private int x=0;
	private int y=0;
	
	public int getX()
	{
		return x;
	}
	
	public int getY()
	{
		return y;
	}

	public PRunnable(PFrame frame, int x, int y)
	{
		this.frame = frame;
		this.x = x;
		this.y = y;
	}
	
	@Override
	public void run()
	{
		while(true)
		{
			try
			{
				//czas uspienia - szybkosc gry (w zaleznosci od levelu)
				Thread.sleep(500);
			}
			catch (InterruptedException e){}

		}
	}
}
