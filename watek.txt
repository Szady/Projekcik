package placyk;

import java.awt.Color;
import java.util.Random;

public class PRunnable implements Runnable
{
	private PFrame frame;
	private int x;
	private int y;
	private Color k;
	
	private int cx;
	private int cy;
	
	private Random rnd = new Random();
	
	public int getX(){return x;}
	
	public int getY(){return y;}
	
	public Color getC(){return k;}

	public PRunnable(PFrame frame, int x, int y)
	{
		this.frame = frame;
		this.x = x;
		this.y = y;
		this.k = new Color(rnd.nextInt(256),rnd.nextInt(256),rnd.nextInt(256));
		
		this.cx = x;
		this.cy = y;
	}
	
	private void move()
	{
		if(x==cx && y==cy)
		{
			cx=rnd.nextInt(400)+30;
			cy=rnd.nextInt(500)+40;
		}
		
		if(x>cx)
			x--;
		else
			x++;
		
		if(y>cy)
			y--;
		else
			y++;
	}
	
	@Override
	public void run()
	{
		
		while(true)
		{
			try{Thread.sleep(75);}catch (InterruptedException e){}
			move();
			frame.repaint();
		}
	}
}
